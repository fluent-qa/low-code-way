root /srv/mediawiki;
index index.html index.htm index.php;

location = / {
	ssi on;
	return 301 /wiki/;
}

location ~ \.php {
	include fastcgi_params;
	fastcgi_split_path_info ^(.+\.php)(.*)$;
	try_files $fastcgi_script_name =404;
	fastcgi_param SCRIPT_FILENAME $document_root$fastcgi_script_name;
	fastcgi_pass unix:/var/run/hhvm/sock;
}

location ^~ /w/languages { deny all; } 
location ^~ /w/maintenance { deny all; } 
location ^~ /w/serialized { deny all; } 
location ~ /.git(/|$) { deny all; }

location ~ ^/wk/(.*) {
	return 301 /wiki/$1;
}

location ~* \.(bmp|gif|jpeg|jpg|png)$ {
	expires 2w;
}

location ^~ /wiki {
	include fastcgi_params;
	fastcgi_param SCRIPT_FILENAME $document_root/w/index.php;
	fastcgi_pass unix:/var/run/hhvm/sock;
}

location ~ ^/skins/(.*) {
	return 302 https://static.orain.org/common/skins/$1;
}


location /server-status {
	stub_status on;
	allow 127.0.0.1;
	deny all;
}

error_page 404 /404.html;
location = /404.html {
	ssi on;
	root /usr/share/nginx/errorpages;
}
error_page 500 /500.html;
location = /500.html {
	ssi on;
	root /usr/share/nginx/errorpages;
}
error_page 502 /502.html;
location = /502.html {
	ssi on;
	root /usr/share/nginx/errorpages;
}

location ~ ^/o/(.*) {
	return 301 /w/index.php?title=Special:UrlRedirector/$1;
}
